

Download fop eg. https://launchpad.net/ubuntu/+source/fop/1:1.0.dfsg2-6/+build/3126005

install:

# sudo dpkg -i Downloads/fop_1.0.dfsg2-6_all.deb Downloads/libfop-java_1.0.dfsg2-6_all.deb
# sudo apt-get install -f
# sudo apt-get install libservlet2.4-java


Font install

# fop-ttfreader FranklinITCBQ-Demi.ttf FranklinITCBQ-Demi.xml

#fop -c fop.xconf ...
